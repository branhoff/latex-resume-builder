\documentclass[a4paper,10pt]{article}           % a4paper sets page size, 10pt sets base font size
\usepackage[utf8]{inputenc}                     % Allows UTF-8 characters (accented letters, etc.)
\usepackage[margin=0.5in, top=0.5in]{geometry}  % Sets page margins - adjust these for tighter/looser spacing
\usepackage{enumitem}                           % Enhanced list formatting - customize bullets, spacing, indentation
\usepackage{hyperref}                           % Makes email and website links clickable
\usepackage{titlesec}                           % Customize section heading formatting and spacing
\usepackage{parskip}                            % Controls paragraph spacing
\usepackage{sectsty}                            % Additional section styling options
\usepackage{charter}                            % Charter font - professional serif font like times, etc.

% Global spacing settings - fine-tune these to adjust overall document density
\setlength{\parindent}{0pt}            % No paragraph indentation (flush left)
\setlength{\parskip}{0pt}              % No extra space between paragraphs

% Section title spacing - controls white space around section headers
% Format: \titlespacing{command}{left spacing}{before spacing}{after spacing}
\titlespacing{\section}{0pt}{3pt plus 1pt minus 1pt}{0pt}      % Main sections (EXPERIENCE, EDUCATION)
\titlespacing{\subsection}{0pt}{2pt plus 1pt minus 1pt}{0pt}   % Job titles, project names

% Section font sizes - keep consistent with body text for compact professional look
\sectionfont{\normalsize}              % Main section headers same size as body text
\subsectionfont{\normalsize}           % Job titles same size as body text

\pagenumbering{gobble}                 % Remove page numbers - standard for resumes
